# bat脚本关闭占用端口号

```bash
@echo off
@echo 请输入程序正在运行的端口号
set /p port=
@echo 将要停止端口号: %port% 下程序的运行
@echo 确认继续吗?
pause
for /f "delims=" %%i in ('netstat -nao^|findstr :%%port%%') do (set res=%%i)
for /f "tokens=4,5 delims= " %%a in ("%res%") do (
	set r=%%a
	set r2=%%b
)
echo 有PID：%r%  %r2%
@echo 停止对应的进程
if defined r (taskkill /pid %r% -t -f)
if defined r2 (taskkill /pid %r2% -t -f)
pause
```
